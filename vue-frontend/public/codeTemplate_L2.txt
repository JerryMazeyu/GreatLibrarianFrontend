def __call__(self, prompt: str, history=None) -> str:
    dashscope.api_key = self.apikey
    response = dashscope.Generation.call(model=dashscope.Generation.Models.qwen_turbo, prompt=prompt, history=history)
    if response:
        if response['output']:
            if response['output']['text']:
                return response['output']['text']
    return 'API Problem'
